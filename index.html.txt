  <script>
    const QUOTES = [
      "The quick brown fox jumps over the lazy dog near the river bank.",
      "Practice makes perfect, and every keystroke brings you closer to mastery.",
      "Speed is nothing without accuracy; learn to type correctly before typing fast.",
      "The best way to predict the future is to invent it yourself.",
      "Consistency is the foundation of all great skills and habits in life.",
      "A journey of a thousand miles begins with a single step forward.",
      "Code is like humor. When you have to explain it, it is bad.",
      "First solve the problem, then write the code as clearly as possible.",
      "Simplicity is the soul of efficiency, so keep it clean and concise.",
      "Learning to type well is one of the most valuable skills you can develop."
    ];
    //variable initialization
    let penalty=0;
    let backspaces = 0;
    let quote     = "";
    let startTime = null;
    let timer     = null;
    let timeLeft  = 20;
    let done      = false;

//element references
    const quoteBox = document.getElementById("quote-box");
    const inputEl  = document.getElementById("input-area");
    const wpmEl    = document.getElementById("wpm");
    const accEl    = document.getElementById("acc");
    const timerEl  = document.getElementById("timer");
    const resultEl = document.getElementById("result-msg");
    const backspaceEl = document.getElementById("backspaces");
//generating quote
    function pickQuote() {
      return QUOTES[Math.floor(Math.random() * QUOTES.length)];
    }
//rendering quote with correct/wrong/pending characters
    function renderQuote(typed) {
      let html = "";
      for (let i = 0; i < quote.length; i++) {
        let cls = "char-pending";
        if (i < typed.length) {
          cls = typed[i] === quote[i] ? "char-correct" : "char-wrong";
        } else if (i === typed.length) {
          cls = "char-pending char-cursor";
        }
        html += `<span class="${cls}">${quote[i]}</span>`;
      }
      quoteBox.innerHTML = html;
    }
    inputEl.addEventListener("keydown", (e) => {
  if (e.key === "Backspace") {
    backspaces++;
    backspaceEl.textContent = backspaces;
  }
});

//WPM calculation
    function calcWpm(typed) {
      const elapsed = (Date.now() - startTime) / 20000;
      const words = typed.trim().split(" ").filter(Boolean).length;
      return elapsed > 0 ? Math.round(words / elapsed) : 0;
    }
//accuracy calculation with penalty for backspaces
   function calcAcc(typed) {
  if (typed.length === 0) return 100;

  let correct = 0;
  for (let i = 0; i < typed.length; i++) {
    if (i < quote.length && typed[i] === quote[i]) correct++;
  }

  let rawAcc = (correct / typed.length) * 100;

  // penalty
  penalty = backspaces * 0.5; // 0.5% penalty per backspace
  return Math.max(0, Math.round(rawAcc - penalty));
}
//test starting function
    function startCountdown() {
      timer = setInterval(() => {
        timeLeft--;
        timerEl.textContent = timeLeft + "s";
        const typed = inputEl.value;
        wpmEl.textContent = calcWpm(typed);
        accEl.textContent = calcAcc(typed) + "%";
        if (timeLeft <= 0) endTest();
      }, 1000);
    }
//test ending function
    function endTest() {
      clearInterval(timer);
      done = true;
      inputEl.disabled = true;
      const typed = inputEl.value;
      wpmEl.textContent   = calcWpm(typed);
      accEl.textContent   = calcAcc(typed) + "%";
      timerEl.textContent = "0s";
      resultEl.textContent = `Done! You typed ${calcWpm(typed)} WPM with ${calcAcc(typed)}% accuracy. Penalty: ${penalty}% for ${backspaces} backspaces.`;
    }
//reinitialising
    function init(newText) {
      clearInterval(timer);
      quote     = newText || pickQuote();
      startTime = null;
      timeLeft  = 20;
      done      = false;
      inputEl.value        = "";
      inputEl.disabled     = false;
      wpmEl.textContent    = "—";
      accEl.textContent    = "—";
      timerEl.textContent  = "20s";
      resultEl.textContent = "";
      renderQuote("");
      inputEl.focus();
       backspaces = 0;
backspaceEl.textContent = "0";
    }

    inputEl.addEventListener("input", () => {
      if (done) return;
      const typed = inputEl.value;
      if (!startTime && typed.length > 0) {
        startTime = Date.now();
        startCountdown();
      }
      renderQuote(typed);
      wpmEl.textContent = startTime ? calcWpm(typed) : "—";
      accEl.textContent = startTime ? calcAcc(typed) + "%" : "—";
      if (typed === quote) endTest();
    });

    document.getElementById("restart-btn").addEventListener("click", () => init(quote));
    document.getElementById("new-btn").addEventListener("click", () => init());

    init();
    
     
    //end of javascript code
  </script>
